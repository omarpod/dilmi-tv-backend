# المكتبات الأساسية لمشروع Dilmi TV Backend

# Django: إطار العمل الأساسي لبناء الموقع/لوحة التحكم

# Django REST Framework: لبناء الـ API الذي يتواصل معه تطبيق الأندرويد
djangorestframework==3.15.1

# django-cors-headers: يسمح لتطبيق الأندرويد (أو أي جهة خارجية)
# بالاتصال بالـ API دون أن يمنعه المتصفح/الخادم بسبب سياسة CORS
django-cors-headers==4.3.1

# Pillow: مطلوبة لحقول الصور (شعارات الفرق، صور الأخبار...)
--only-binary=:all: Pillow

# (صفحتي "سياسة الخصوصية" و"من نحن")، يسمح بتنسيق النص من لوحة التحكم
# دون كتابة وسوم HTML يدوياً
django-ckeditor-5==0.2.13

# whitenoise: يسمح لـ Django/Gunicorn بخدمة ملفات CSS/JS الثابتة مباشرة

# لخادم ويب منفصل مثل Nginx — الحل القياسي على منصات مثل Render/Heroku
whitenoise==6.7.0

# جزء (host, port, user, password...) يدوياً في settings.py
dj-database-url==2.2.0

# psycopg: المُشغّل (driver) الفعلي الذي يتيح لـ Django التحدث فعلياً
# مع قاعدة بيانات PostgreSQL/Neon. نسخة [binary] لا تتطلب أدوات ترجمة
# (compiler) إضافية أثناء التثبيت، وهي الأنسب لبيئات النشر السحابية.
psycopg[binary]==3.2.1
